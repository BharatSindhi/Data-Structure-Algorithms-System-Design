✅ Your Code (Reference)
void bubbleSort(int arr[], int n) { // O(n^2)
    for (int i = 0; i < n - 1; i++) {
        bool isSwap = false;

        for (int j = 0; j < n - i - 1; j++) {
            if (arr[j] > arr[j + 1]) {
                swap(arr[j], arr[j + 1]);
                isSwap = true;
            }
        }

        if (!isSwap) {
            return;   // array already sorted
        }
    }
}
